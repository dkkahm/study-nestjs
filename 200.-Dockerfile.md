FROM node:latest as builder
WORKDIR '/app'
COPY package.json .
RUN npm install
COPY ./tsconfig.json .
COPY ./tsconfig.build.json .
COPY ./nest-cli.json .
COPY ./src ./src
RUN npm run build

FROM node:alpine
WORKDIR '/app'
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/dist ./dist
COPY package.json .

EXPOSE 3000

CMD ["node", "dist/main"]
